배열의 활용(1)#

#include <stdio.h>
int main()
{
    int score[6];
    int i, a,
     sum = 0;
 
    for (i = 0; i < 6; i++) {   // 배열과 반복문을 사용하여 scanf() 함수로 입력 받음
        printf("학생 %d 성적을 입력하세요 : ", i + 1);  // 학생은 1번부터 시작하므로 i + 1을 사용함
        scanf("%d", &score[i]); // score[0]의 값부터 입력
    }
 
    for (i = 0; i < 6; i++)
        sum += score[i];    // sum = score[0] + score[1] + ••• + score[5]
 
        printf("합계 : %d\n", sum);
        printf("평균 : %.2lf\n", (double)sum/6);    // 평균이 실수형으로 계산되도록 (double)을 사용하여 형 변환
        printf("\n score[6]에 저장되어있는 숫자:\n");
 
    for(a=0;a<6;a++)
        printf(" %d",score[a]);
 
    return 0;
}

학생 1 성적을 입력하세요 : 100 학생 2 성적을 입력하세요 : 90 학생 3 성적을 입력하세요 : 80 학생 4 성적을 입력하세요 : 70 학생 5 성적을 입력하세요 : 60 학생 6 성적을 입력하세요 : 50 합계 : 450 평균 : 90.00

score[6]에 저장되어있는 숫자: 100 90 80 70 60 50
